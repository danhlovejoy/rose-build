# Display helper — use this throughout the lab
def show_images(images, titles, cols=3, figsize=(15, 5)):
    """Display a row of images with titles."""
    rows = (len(images) + cols - 1) // cols
    fig, axes = plt.subplots(rows, cols, figsize=(figsize[0], figsize[1] * rows))
    if rows == 1:
        axes = [axes] if cols == 1 else axes
    else:
        axes = axes.flatten()
    for i, (img, title) in enumerate(zip(images, titles)):
        if len(img.shape) == 2:  # grayscale
            axes[i].imshow(img, cmap='gray')
        else:  # color
            axes[i].imshow(cv2.cvtColor(img, cv2.COLOR_BGR2RGB))
        axes[i].set_title(title, fontsize=10)
        axes[i].axis('off')
    for j in range(i + 1, len(axes)):
        axes[j].axis('off')
    plt.tight_layout()
    plt.show()

Part 1: Blurring Filters

Blurring smooths out variation between neighboring pixels. Each filter does this differently. Your job is to apply all three, display the results, and explain what's happening.

# Cell 4: Apply three blur filters
# Box blur — every pixel in the kernel has equal weight
box = cv2.blur(img, (7, 7))

# Gaussian blur — center pixel gets more weight, falls off like a bell curve
gaussian = cv2.GaussianBlur(img, (7, 7), 0)

# Median blur — replaces each pixel with the median of its neighbors
# Especially good at removing "salt and pepper" noise
median = cv2.medianBlur(img, 7)

show_images(
    [img, box, gaussian, median],
    ["Original", "Box Blur (7×7)", "Gaussian Blur (7×7)", "Median Blur (7)"],
    cols=4, figsize=(18, 5)
)

Part 2: Edge Detection

Edge detection finds places where pixel values change sharply — that's where the boundaries of objects are. You'll use two approaches: the Sobel operator (which detects edges in one direction at a time) and Canny (which combines multiple steps into a clean edge map).

# Cell 5: Sobel edge detection
# Sobel in the X direction — detects vertical edges
sobel_x = cv2.Sobel(gray, cv2.CV_64F, 1, 0, ksize=3)
sobel_x_abs = np.uint8(np.absolute(sobel_x))

# Sobel in the Y direction — detects horizontal edges
sobel_y = cv2.Sobel(gray, cv2.CV_64F, 0, 1, ksize=3)
sobel_y_abs = np.uint8(np.absolute(sobel_y))

# Combined magnitude
sobel_combined = cv2.bitwise_or(sobel_x_abs, sobel_y_abs)

show_images(
    [gray, sobel_x_abs, sobel_y_abs, sobel_combined],
    ["Grayscale", "Sobel X (vertical edges)", "Sobel Y (horizontal edges)", "Sobel Combined"],
    cols=4, figsize=(18, 5)
)
# Cell 6: Canny edge detection
# Canny uses two thresholds to decide what counts as an "edge"
# Low threshold: weak edges below this are discarded
# High threshold: strong edges above this are kept
# Edges between the two are kept only if connected to a strong edge

canny = cv2.Canny(gray, threshold1=50, threshold2=150)

show_images(
    [gray, sobel_combined, canny],
    ["Grayscale", "Sobel Combined", "Canny (50, 150)"],
    cols=3, figsize=(15, 5)
)
Markdown cell after Cell 6: Compare Sobel X, Sobel Y, and the combined result. Why does the X kernel find vertical edges (not horizontal)? What does Canny give you that raw Sobel doesn't? Look at the Computerphile "Finding the Edges" video if you need a refresher on how Sobel works mathematically.

Part 3: Experiment with Kernel Size

The size of the kernel changes the filter's behavior dramatically. A small kernel detects fine detail. A large kernel smooths aggressively. Your job is to see this firsthand.

# Cell 7: Kernel size comparison — Gaussian blur
sizes = [3, 7, 15, 31]
blurred = [cv2.GaussianBlur(img, (s, s), 0) for s in sizes]

show_images(
    [img] + blurred,
    ["Original"] + [f"Gaussian {s}×{s}" for s in sizes],
    cols=5, figsize=(20, 5)
)
# Cell 8: Kernel size comparison — Canny edge detection
# Blur first with different kernel sizes, then apply Canny
# This shows how preprocessing affects edge detection

edge_results = []
edge_labels = []
for s in [0, 3, 7, 15]:
    if s == 0:
        blurred_gray = gray  # no blur
        label = "Canny (no blur)"
    else:
        blurred_gray = cv2.GaussianBlur(gray, (s, s), 0)
        label = f"Blur {s}×{s} → Canny"
    edges = cv2.Canny(blurred_gray, 50, 150)
    edge_results.append(edges)
    edge_labels.append(label)

show_images(edge_results, edge_labels, cols=4, figsize=(18, 5))

What Makes This Lab Interesting

Spatial filtering is one of the oldest ideas in computer vision, and it's still everywhere. The Sobel operator was published in 1968. Canny edge detection is from 1986. But the concept of sliding a small kernel over an array and computing a weighted sum at each position — that's convolution, and it's the same fundamental operation that powers modern convolutional neural networks (CNNs). When you apply a 3×3 Sobel kernel to your image, you're doing exactly what the first layer of a deep learning model does, except the model learns its own kernels from data instead of using hand-designed ones.

That's where this course is heading: from hand-designed filters to learned filters. But you have to understand what a kernel does before you can appreciate what it means for a network to learn one.
