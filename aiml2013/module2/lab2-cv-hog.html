Module 2 Lab: HOG Features & Image Similarity

Build a visual feature extractor, then find where it breaks.

The Task

In the readings and in class, you learned that edge detection kernels find boundaries by measuring how quickly pixel values change across a small neighborhood. This week you will build on that foundation: instead of applying a single edge-detection kernel to visualize one image, you will extract a compact feature descriptor from every image in a dataset and use those descriptors to retrieve visually similar images.

The descriptor is called HOG (Histogram of Oriented Gradients). It was state-of-the-art for pedestrian detection in the mid-2000s and is still used in production systems today. It works by dividing an image into small cells, computing the gradient direction and magnitude in each cell, and building a histogram of those orientations. The result is a fixed-length vector that encodes the local edge structure of the image in a way that is partially robust to small changes in lighting or position.

You will build HOG descriptors, use them for similarity retrieval, and systematically find where the approach fails. Those failures are the lab. They tell you what HOG can and cannot represent, and that is exactly what you need to know before next week.

# Cell 3: Compute HOG feature vectors
# HOG divides the image into a grid of cells, computes gradient histograms
# in each cell, normalizes them over blocks of cells, and concatenates
# everything into one flat vector.

# If using your own images (Cell 2B), increase TARGET_SIZE to 128 for richer features.
TARGET_SIZE = 64

def image_to_hog(image):
    """Convert a color image to a HOG feature vector and a visualization image."""
    img  = transform.resize(np.array(image), (TARGET_SIZE, TARGET_SIZE),
                            anti_aliasing=True)
    gray = rgb2gray(img)
    features, hog_vis = hog(
        gray,
        orientations=8,           # 8 gradient direction bins (0°–180°)
        pixels_per_cell=(8, 8),   # each cell covers an 8×8 pixel patch
        cells_per_block=(2, 2),   # normalize over 2×2 blocks of cells
        visualize=True
    )
    return features, hog_vis

hog_features, hog_visuals = [], []
for img in images:
    feats, vis = image_to_hog(img)
    hog_features.append(feats)
    hog_visuals.append(vis)

hog_matrix = np.array(hog_features)
print(f"HOG feature matrix:  {hog_matrix.shape}")
print(f"  {hog_matrix.shape[0]} images")
print(f"  {hog_matrix.shape[1]} features per image")
print(f"\nSparsity: {100*(hog_matrix == 0).mean():.1f}% of values are zero")
print(f"  (HOG is relatively dense compared to bag-of-words text features)")

Part 3: Similarity Retrieval

# Cell 6: Find visually similar images using HOG cosine similarity
def find_similar(query_idx, n=5):
    """Return n most similar images by HOG cosine similarity."""
    query_vec = hog_matrix[query_idx:query_idx + 1]
    scores = cosine_similarity(query_vec, hog_matrix)[0]
    scores[query_idx] = -1   # exclude the query itself
    top_idx = scores.argsort()[-n:][::-1]
    return top_idx, scores[top_idx]


N_PER_CLASS = len(images) // len(set(img_names))
query_indices = [0, N_PER_CLASS, N_PER_CLASS * 2]

for q_idx in query_indices:
    similar_idx, sim_scores = find_similar(q_idx)

    fig, axes = plt.subplots(1, 6, figsize=(15, 3))
    axes[0].imshow(images[q_idx])
    axes[0].set_title(f'QUERY\n{img_names[q_idx]}', fontweight='bold', fontsize=9)
    axes[0].axis('off')

    for i, (idx, score) in enumerate(zip(similar_idx[:5], sim_scores[:5])):
        correct = (img_names[idx] == img_names[q_idx])
        marker = "\u2713" if correct else "\u2717"
        axes[i + 1].imshow(images[idx])
        axes[i + 1].set_title(f'{marker} {img_names[idx]}\ncos={score:.3f}', fontsize=8)
        axes[i + 1].axis('off')

    plt.tight_layout()
    plt.show()

# Cell 7: Systematic failure analysis
n_queries = min(40, len(images))
failures = []

for q_idx in range(n_queries):
    similar_idx, sim_scores = find_similar(q_idx, n=3)
    top = similar_idx[0]
    if img_names[top] != img_names[q_idx]:
        failures.append({
            'query_idx':   q_idx,
            'query_class': img_names[q_idx],
            'match_idx':   top,
            'match_class': img_names[top],
            'score':       sim_scores[0]
        })

print(f"HOG top-1 failures: {len(failures)} out of {n_queries} queries "
      f"({100*len(failures)/n_queries:.0f}% error rate)\n")
print("Class pairs most often confused:")
from collections import Counter
pair_counts = Counter((f['query_class'], f['match_class']) for f in failures)
for (q_cls, m_cls), count in pair_counts.most_common(5):
    print(f"  {q_cls:<14} matched to  {m_cls:<14}  ({count}x)")

# Visualize the most interesting failure
if failures:
    f = failures[0]

Part 4: What HOG Knows and Doesn’t Know

# Cell 8: Compare retrieval accuracy across classes
from collections import defaultdict

class_correct = defaultdict(lambda: {'correct': 0, 'total': 0})

for q_idx in range(len(images)):
    similar_idx, _ = find_similar(q_idx, n=1)
    top = similar_idx[0]
    cls = img_names[q_idx]
    class_correct[cls]['total'] += 1
    if img_names[top] == cls:
        class_correct[cls]['correct'] += 1

print(f"{'Class':<16}  {'Top-1 Accuracy':>15}")
print("-" * 34)
sorted_classes = sorted(class_correct.items(),
                        key=lambda x: x[1]['correct']/x[1]['total'],
                        reverse=True)
for cls, counts in sorted_classes:
    acc = counts['correct'] / counts['total']
    bar = '\u2588' * int(acc * 20)
    print(f"{cls:<16}  {acc:6.0%}  {bar}")

print()

best_cls  = sorted_classes[0][0]
worst_cls = sorted_classes[-1][0]
print(f"Easiest class for HOG:  {best_cls}")
print(f"Hardest class for HOG:  {worst_cls}")
